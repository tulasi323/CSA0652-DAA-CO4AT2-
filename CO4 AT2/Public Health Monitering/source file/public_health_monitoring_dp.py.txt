# Smart Public Health Monitoring using Dynamic Programming

n = int(input("Enter number of records: "))

cases = list(map(int, input("Enter disease cases: ").split()))

dp = [0] * n

dp[0] = cases[0]

for i in range(1, n):
    dp[i] = dp[i-1] + cases[i]

print("Disease Cases =", cases)
print("Predicted Total Cases =", dp[n-1])
